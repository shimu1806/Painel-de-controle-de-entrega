// Função para filtrar os dados com base nas checkboxes selecionadas
function filterByStatus() {
    const selectedStatuses = Array.from(document.querySelectorAll('.filter-box:checked')).map(checkbox => checkbox.value)
    const filteredData = tableData.filter(item => selectedStatuses.includes(item.CB7_STATUS))
    displayFilteredData(filteredData)
    console.log("filteredData")
}
